

use math: sqrt

function mean(a)
  return a.sum()/size(a)
end

function sigma(a,m=null)
  if m is null
    m = mean(a)
  end
  return sqrt(a.sum(|x| (x-m)^2)/(size(a)-1))
end

Stat = table{
  function string
    """\
    mean  = {},\n\
    sigma = {}\
    """ % [str(self.mean,"f",4), str(self.sigma,"f",4)]
  end
}

function stat(a)
  m = mean(a)
  return table Stat{mean = m, sigma = sigma(a,m)}
end

s = stat(rand(1..6).list(10000))

print(s)
