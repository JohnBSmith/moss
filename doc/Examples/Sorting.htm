Mergesort

msort = |a| (copy(a) if size(a)<=1 else
  merge(msort(a[1..:2]),msort(a[0..:2])))

function merge(L,R)
  y = []
  while size(L)!=0 and size(R)!=0
    y.push((L if L[-1]>R[-1] else R).pop())
  end
  return L+R+y.rev()
end
