# A simple scanner:
# takes an input string,
# returns a list of tokens.

SyntaxError = table{}

function scan(s)
  a = []
  n = size(s)
  i = 0
  while i<n
    if s[i] in "+-*/^(){}[];,"
      a.push(s[i])
      i+=1
    elif s[i].isalpha()
      j = i
      while i<n and s[i].isalnum()
        i+=1
      end
      a.push(s[j..i-1])
    elif s[i].isdigit()
      j = i
      while i<n and s[i].isdigit()
        i+=1
      end
      a.push(s[j..i-1])
    elif s[i].isspace()
      i+=1
    else
      raise table SyntaxError{
        value = "unexpected character: '{}'"%[s[i]]
      }
    end
  end
  return a
end

while true
  s = input("> ")
  try
    print(scan(s))
  catch e if e: SyntaxError
    print("Syntax error: ", e.value, ".")
  end
end
