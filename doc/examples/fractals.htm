use gx: canvas, sleep

function fractal(f,z0,argm={})
  {w=720, h=480, n=40} = argm
  cv = canvas(w,h)
  for y in h
    for x in w
      c = (4*x/h-3)-(4*y/h-2)*1i
      z = z0(c)
      for i in n
        z = f(z,c)
        if abs(z)>2
          cv.hsl(0,0,1-i/n)
          cv.fill(x,y,1,1)
          break
        end
      end
    end
    cv.flush()
  end
  while cv.key()!="q"
    sleep(0.1)
  end
end

# Mandelbrot set
fractal(|z,c| z^2+c, |c| 0)

# Julia set of f(z) = z^2-0.6+0.5i
fractal(|z,c| z^2-0.6+0.5i, |c| c, n=80)
